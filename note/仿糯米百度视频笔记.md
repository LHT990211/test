      public function getFirstCategorys($parentId = 0) {
              $data = [
                  'parent_id' => $parentId,
                  'status' => ['neq',-1],
              ];
              //想根据那个属性来排序
              //可以直接在这里设置属性 
              $order =[
                  'listorder' => 'desc',
                  'id' => 'desc',
              ];
              $result = $this->where($data)
                  ->order($order)
                  ->paginate();
              echo $this->getLastSql();
      
              return $result;
      
          }
